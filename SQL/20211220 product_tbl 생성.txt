create table product_tbl(
p_number number not null,
p_type varchar2(10) not null,
p_name varchar2(20) not null,
p_price number not null,
p_information varchar2(2000) not null,
p_mainphoto varchar2(50) not null,
p_photo1 varchar2(50) not null,
p_photo2 varchar2(50) not null,
p_photo3 varchar2(50) not null,
p_date date default sysdate,
p_state VARCHAR2(20)    default 'true'  not null,
primary key(p_number)
);

create sequence product_tbl_seq
start with 1
increment by 1;

commit;
